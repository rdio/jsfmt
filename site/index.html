jsfmt

Like gofmt, but for javascript

Formatting

					
# Format your code and write it to disk
jsfmt -w=true source.js

# View a diff of all formatting changes without writing them
jsfmt -d=true source.js

# List all files that need formatting in the `src/` directory
jsfmt -l=true -w=false src/

Searching

					
var jsfmt = require('jsfmt');
var fs = require('fs');

var js = fs.readFileSync('my_file.js');

jsfmt.search(js, "R.Component.create(a, { dependencies: z })").forEach(function(matches, wildcards) {
  console.log(wildcards.z);
});

Replacing

					
var jsfmt = require('jsfmt');
var fs = require('fs');

var js = fs.readFileSync('my_file.js');

js = jsfmt.rewrite(js, "_.each(a, b) -> a.forEach(b)");
