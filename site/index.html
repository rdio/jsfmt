jsfmt

Jsfmt formats, searches, and rewrites javascript.

Formatting

					
# Format `source.js` and print it to stdout.
jsfmt source.js

# Format `source.js` and write it to disk.
jsfmt -w=true source.js

# View a diff of all formatting changes in `source.js` without writing them.
jsfmt -d=true source.js

# List all files that need formatting in the `src/` directory.
jsfmt -l=true -w=false src/

Searching

					
# Find all instances of underscore.js's reduce in `source.js`. Note that spacing doesn't matter.
jsfmt --search="_.reduce(a, b, c)"

# Find all calls to console.log() in your project's `src/` directory.
jsfmt --search="console.log(s)"

Replacing

					
# Replace underscore.js's reduce with the native reduce method in `source.js`.
jsfmt --rewrite="_.reduce(a, b, c) -> a.reduce(b, c)" examples/reduce.js

# Commit the above changes to disk
jsfmt --write=true --rewrite="_.reduce(a, b, c) -> a.reduce(b, c)" examples/reduce.js

# Replace modulo with an expression that always returns a positive number, and run the results.
jsfmt --rewrite="x % y -> ((x % y) + y) % y" source.js | node
